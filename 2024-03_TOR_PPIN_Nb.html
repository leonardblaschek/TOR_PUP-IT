## Loading function
read_data <- function(flnm) {
  read_tsv(flnm) |>
    mutate(batch = str_remove(
      basename(flnm),
      fixed(str_remove(suffix, fixed("*")))
    )) %>%
    # base pipe does not support curly brackets
    # filter out contaminants and decoys
    {
      if (input == "FP") {
        filter(., str_detect(!!protein_name, "ARATH|FLAG-PUP|PafA|GFP|NbE|NbD|RAPTOR|LST8"))
      } else {
        filter(., !str_detect(!!protein_name, "REV_|CON_"))
      }
    } |>
    select(
      quo_name(seq_name),
      batch,
      !!int_selector & !contains(intensity_cols[2])
    ) |>
    ## Tidy data frame to one row per value
    pivot_longer(!!int_selector,
      names_to = "ID",
      values_to = "Intensity"
    ) |>
    mutate(ID = str_remove(ID, intensity_cols[1])) %>% # base pipe does not support curly brackets
    {
      if (input == "MQ") {
        left_join(., map_dfr(meta_files, read_tsv)) |> mutate(ID = sample)
      } else {
        .
      }
    } |>
    ## Clean column names and replace missing values (0 in the input) with NA
    mutate(
      sample = ID,
      group = str_remove(sample, "_[:digit:]$"),
      Intensity = case_when(
        Intensity == 0 ~ NA_real_,
        TRUE ~ Intensity
      )
    )
}

      names(pep_df) %in%
        names(pep_df |>
          select(!!int_selector))
    )

    pe <- readQFeatures(
      pep_df,
      ecol = ecols,
      name = "peptideNorm", sep = "\t"
    )

    # Load metadata
    colData(pe)$group <- colnames(pe)[[1]] |>
      str_remove(str_replace_all(intensity_cols[1], " ", fixed("."))) |>
      str_remove("_[:digit:]$")

    ## Overview on peptide level
    plotMDS(assay(pe[["peptideNorm"]]), main = "Peptide level")

    ## Summarise to protein level
    pe <- aggregateFeatures(pe,
      i = "peptideNorm",
      fcol = quo_name(protein_name_dot),
      name = "protein",
      maxit = 500
    )

    plotMDS(assay(pe[["protein"]]), main = "Protein level")

    ## Set aside protein intensities
    protein_int <- as_tibble(assay(pe[["protein"]]), rownames = "protein") |>
      pivot_longer(
        -protein,
        names_to = "group",
        values_to = "intensity"
      ) |>
      mutate(
        group = str_remove(group, "\\..+"),
        batch = str_extract(basename(data_files[i]), "(R[:digit:]{4})\\.tsv", group = 1)
      ) |>
      separate_wider_delim(
        group,
        delim = "_",
        names = c("species", "construct", "bait", "replicate")
      )

    intensities <- bind_rows(intensities, protein_int)

    ## Estimate protein levels
    pe <- msqrob(
      object = pe,
      i = "protein",
      formula = ~group,
      overwrite = TRUE,
      maxitRob = 1000
    )

    ## Compare groups
    coefs <- names(getCoef(rowData(pe[["protein"]])$msqrobModels[[1]]))[-1]
    coef_comb_data <- expand.grid(coefs, coefs) |>
      filter(Var1 != Var2)

    indx <- !duplicated(t(apply(coef_comb_data, 1, sort)))

    coef_comb <- coef_comb_data[indx, ] |>
      unite(contrast, Var1, Var2, sep = " - ") |>
      pull(var = contrast, name = NULL)

    contrast_names <- c(coefs, coef_comb)
    contrast_list <- paste0(c(coefs, coef_comb), "=0")

    contrasts <- makeContrast(
      contrast_list,
      parameterNames = coefs
    )

    pe <- hypothesisTest(object = pe, i = "protein", contrast = contrasts, overwrite = TRUE)

    for (k in 1:length(contrast_names)) {
      ex <- as_tibble(rowData(pe[["protein"]])[[contrast_names[k]]], rownames = "protein") |>
        mutate(
          contrast = str_remove_all(contrast_names[k], "group"),
          batch = str_remove(basename(data_files[i]), fixed(".tsv"))
        ) |>
        separate(contrast, into = c("group1", "group2"), sep = " - ", fill = "right") |>
        mutate(group2 = case_when(
          is.na(group2) ~ unique(sort(colData(pe)$group))[1],
          TRUE ~ group2
        ))

      FC <- FC |>
        bind_rows(ex)
    }
  }

  ## Export data

  # Export protein intensity data
  write_tsv(
    intensities,
    paste0(out_path, input, "_maxLFQ-", maxLFQ, "_hurdle-", hurdle, "_glMinProb-", global_MinProb, "_protein_levels.tsv")
  )

  # Export differential enrichment
  write_tsv(
    FC,
    paste0(out_path, input, "_maxLFQ-", maxLFQ, "_hurdle-", hurdle, "_glMinProb-", global_MinProb, "_PPI_FC.tsv")
  )

  xlsx_export <- FC |>
    arrange(group1, group2, desc(logFC)) |>
    group_by(group1, group2, batch) |>
    group_split()
  writexl::write_xlsx(
    xlsx_export,
    paste0(out_path, input, "_maxLFQ-", maxLFQ, "_hurdle-", hurdle, "_glMinProb-", global_MinProb, "_PPI_FC.xlsx")
  )
